67 void BESDDSResponseHandler::execute(BESDataHandlerInterface &dhi)
68 {
69  // NOTE: It is the responsibility of the specific request handler to set
70  // the BaseTypeFactory. It is set to NULL here
71  dhi.action_name = DDS_RESPONSE_STR;
72  DDS *dds = new DDS(NULL, "virtual");
73  BESDDSResponse *bdds = new BESDDSResponse(dds);
74 
75  // Set the DAP protocol version requested by the client
76 
77  dhi.first_container();
78  BESDEBUG("version", "Initial CE: " << dhi.container->get_constraint() << endl);
79 
80  // Keywords were a hack to the protocol and have been dropped. We can get rid of
81  // this keyword code. jhrg 11/6/13
82  dhi.container->set_constraint(dds->get_keywords().parse_keywords(dhi.container->get_constraint()));
83  BESDEBUG("version", "CE after keyword processing: " << dhi.container->get_constraint() << endl);
84 
85  if (dds->get_keywords().has_keyword("dap")) {
86  dds->set_dap_version(dds->get_keywords().get_keyword_value("dap"));
87  }
88  else if (!bdds->get_dap_client_protocol().empty()) {
89  dds->set_dap_version(bdds->get_dap_client_protocol());
90  }
91 
92  _response = bdds;
93 
94  BESRequestHandlerList::TheList()->execute_each(dhi);
95 }
