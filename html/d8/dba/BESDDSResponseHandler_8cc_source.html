74 void BESDDSResponseHandler::execute(BESDataHandlerInterface &dhi)
75 {
76  dhi.action_name = DDS_RESPONSE_STR;
77 
78  GlobalMetadataStore *mds = GlobalMetadataStore::get_instance();
79 
80  GlobalMetadataStore::MDSReadLock lock;
81 
82  dhi.first_container();
83  if (mds) lock = mds->is_dds_available(dhi.container->get_real_name());
84 
85  if (mds && lock() && dhi.container->get_constraint().empty()) {
86  // FIXME Does not work for constrained DDS requests
87  // send the stored response
88  mds->get_dds_response(dhi.container->get_real_name(), dhi.get_output_stream());
89  // suppress transmitting a ResponseObject in transmit()
90  d_response_object = 0;
91  }
92  else {
93  DDS *dds = new DDS(NULL, "virtual");
94 
95 #if 0
96  // Keywords were a hack to the protocol and have been dropped. We can get rid of
97  // this keyword code. jhrg 11/6/13
98  dhi.container->set_constraint(dds->get_keywords().parse_keywords(dhi.container->get_constraint()));
99 
100  if (dds->get_keywords().has_keyword("dap")) {
101  dds->set_dap_version(dds->get_keywords().get_keyword_value("dap"));
102  }
103  else if (!bdds->get_dap_client_protocol().empty()) {
104  dds->set_dap_version(bdds->get_dap_client_protocol());
105  }
106 #endif
107 
108  d_response_object = new BESDDSResponse(dds);
109 
110  BESRequestHandlerList::TheList()->execute_each(dhi);
111 
112  if (mds) {
113  dhi.first_container(); // must reset container; execute_each() iterates over all of them
114  mds->add_responses(static_cast<BESDDSResponse*>(d_response_object)->get_dds(),
115  dhi.container->get_real_name());
116  }
117  }
118 }
