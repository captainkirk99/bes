70 void
71 BESVersionResponseHandler::execute( BESDataHandlerInterface &dhi )
72 {
73  BESVersionInfo *info = new BESVersionInfo() ;
74  _response = info ;
75  dhi.action_name = VERS_RESPONSE_STR ;
76  info->begin_response( VERS_RESPONSE_STR, dhi ) ;
77 
78  string administrator = "" ;
79  try
80  {
81  bool found = false ;
82  vector<string> vals ;
83  string key = "BES.ServerAdministrator" ;
84  TheBESKeys::TheKeys()->get_value( key, administrator, found ) ;
85  }
86  catch( ... )
87  {
88  administrator = DEFAULT_ADMINISTRATOR ;
89  }
90  if( administrator.empty() )
91  {
92  administrator = DEFAULT_ADMINISTRATOR ;
93  }
94  info->add_tag( "Administrator", administrator ) ;
95 
96  info->add_library( PACKAGE_NAME, PACKAGE_VERSION ) ;
97 
98  BESRequestHandlerList::TheList()->execute_all( dhi ) ;
99 
100  info->end_response() ;
101 }
