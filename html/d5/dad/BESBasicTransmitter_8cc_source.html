64 void BESTransmitter::conditional_timeout_cancel()
65 {
66  bool cancel_timeout_on_send = false;
67  bool found = false;
68  string doset ="";
69  const string dosettrue ="true";
70  const string dosetyes = "yes";
71 
72  TheBESKeys::TheKeys()->get_value( BES_KEY_TIMEOUT_CANCEL, doset, found ) ;
73  if( true == found ) {
74  doset = BESUtil::lowercase( doset ) ;
75  if( dosettrue == doset || dosetyes == doset )
76  cancel_timeout_on_send = true;
77  }
