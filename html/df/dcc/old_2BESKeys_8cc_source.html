89 void TheBESKeys::initialize_keys()
90 {
91  _keys_file = new ifstream(_keys_file_name.c_str());
92 
93  if (!(*_keys_file)) {
94  char path[500];
95  getcwd(path, sizeof(path));
96  string s = string("BES: fatal, cannot open BES configuration file ") + _keys_file_name + ": ";
97  char *err = strerror(errno);
98  if (err)
99  s += err;
100  else
101  s += "Unknown error";
102 
103  s += (string) ".\n" + "The current working directory is " + path;
104  throw BESInternalFatalError(s, __FILE__, __LINE__);
105  }
106 
107  try {
108  load_keys();
109  }
110  catch (BESError &e) {
111  // be sure we're throwing a fatal error, since the BES can't run
112  // within the configuration file
113  clean();
114  throw BESInternalFatalError(e.get_message(), e.get_file(), e.get_line());
115  }
116  catch (...) {
117  clean();
118  string s = (string) "Undefined exception while trying to load keys from bes configuration file "
119  + _keys_file_name;
120  throw BESInternalFatalError(s, __FILE__, __LINE__);
121  }
122 }

156 void TheBESKeys::load_keys()
157 {
158  string key, value, line;
159  while (!_keys_file->eof()) {
160  bool addto = false;
161  getline(*_keys_file, line);
162  if (break_pair(line.c_str(), key, value, addto)) {
163  if (key == BES_INCLUDE_KEY) {
164  // We make this call to set_key() and force 'addto' to
165  // be true because we need access to the child configuration
166  // files and their values for the admin interface.
167  set_key(key, value, true);
168  load_include_files(value);
169  }
170  else {
171  set_key(key, value, addto);
172  }
173  }
174  }
175 }

177 // The string contained in the character buffer b should be of the
178 // format key=value or key+=value. The pair is broken apart, storing the
179 // key in the key parameter and the value of the key in the value
180 // parameter. If += is used, then the value should be added to the value
181 // of key, not replacing.
182 //
183 // It used to be that we would validate the key=value line. Instead,
184 // anything after the equal sign is considered the value of the key.
185 inline bool TheBESKeys::break_pair(const char* b, string& key, string &value, bool &addto)
186 {
187  addto = false;
188  // Ignore comments and lines with only spaces
189  if (b && (b[0] != '#') && (!only_blanks(b))) {
190  register size_t l = strlen(b);
191  if (l > 1) {
192  int pos = 0;
193  bool done = false;
194  for (register size_t j = 0; j < l && !done; j++) {
195  if (b[j] == '=') {
196  if (!addto)
197  pos = j;
198  else {
199  if (pos != static_cast<int>(j - 1)) {
200  string s = string("BES: Invalid entry ") + b + " in configuration file " + _keys_file_name
201  + " '+' character found in variable name" + " or attempting '+=' with space"
202  + " between the characters.\n";
203  throw BESInternalFatalError(s, __FILE__, __LINE__);
204  }
205  }
206  done = true;
207  }
208  else if (b[j] == '+') {
209  addto = true;
210  pos = j;
211  }
212  }
213  if (!done) {
214  string s = string("BES: Invalid entry ") + b + " in configuration file " + _keys_file_name + ": "
215  + " '=' character not found.\n";
216  throw BESInternalFatalError(s, __FILE__, __LINE__);
217  }
218 
219  string s = b;
220  key = s.substr(0, pos);
221  BESUtil::removeLeadingAndTrailingBlanks(key);
222  if (addto)
223  value = s.substr(pos + 2, s.size());
224  else
225  value = s.substr(pos + 1, s.size());
226  BESUtil::removeLeadingAndTrailingBlanks(value);
227 
228  return true;
229  }
230 
231  return false;
232  }
233 
234  return false;
235 }

246 void TheBESKeys::load_include_files(const string &files)
247 {
248  string newdir;
249  BESFSFile allfiles(files);
250 
251  // If the files specified begin with a /, then use that directory
252  // instead of the current keys file directory.
253  if (!files.empty() && files[0] == '/') {
254  newdir = allfiles.getDirName();
255  }
256  else {
257  // determine the directory of the current keys file. All included
258  // files will be relative to this file.
259  BESFSFile currfile(_keys_file_name);
260  string currdir = currfile.getDirName();
261 
262  string alldir = allfiles.getDirName();
263 
264  if ((currdir == "./" || currdir == ".") && (alldir == "./" || alldir == ".")) {
265  newdir = "./";
266  }
267  else {
268  if (alldir == "./" || alldir == ".") {
269  newdir = currdir;
270  }
271  else {
272  newdir = currdir + "/" + alldir;
273  }
274  }
275  }
276 
277  // load the files one at a time. If the directory doesn't exist,
278  // then don't load any configuration files
279  BESFSDir fsd(newdir, allfiles.getFileName());
280  BESFSDir::fileIterator i = fsd.beginOfFileList();
281  BESFSDir::fileIterator e = fsd.endOfFileList();
282  for (; i != e; i++) {
283  load_include_file((*i).getFullPath());
284  }
285 }
