59 void SiteMapCommand::parse_request(xmlNode *node)
60 {
61  string name;
62  string value;
63  map<string, string> props;
64  BESXMLUtils::GetNodeInfo(node, name, value, props);
65  if (name != SITE_MAP_STR) { // buildSiteMap
66  throw BESSyntaxUserError(string("The specified command ") + name + " is not a build site map command", __FILE__, __LINE__);
67  }
68 
69  string prefix = props["prefix"];
70  string node_suffix = props["nodeSuffix"];
71  string leaf_suffix = props["leafSuffix"];
72 
73  string catalog_name = props["catalog"].empty() ? BESCatalogList::TheCatalogList()->default_catalog_name(): props["catalog"];
74  string filename = props["filename"].empty() ? "site_map.txt": props["filename"];
75 
76  BESDEBUG("besxml", "In SiteMapCommand::parse_request, command attributes: " << prefix << ", "
77  << node_suffix << ", " << leaf_suffix << ", " << catalog_name << ", " << filename << endl);
78 
79  if (prefix.empty() || catalog_name.empty() || filename.empty() || (node_suffix.empty() && leaf_suffix.empty()))
80  throw BESSyntaxUserError("Build site map must include the prefix and at least one of the nodeSuffix or leafSuffix attributes.", __FILE__, __LINE__);
81 
82  BESCatalog *catalog = BESCatalogList::TheCatalogList()->find_catalog(catalog_name);
83  if (!catalog)
84  throw BESSyntaxUserError(string("Build site map could not find the catalog: ") + catalog_name, __FILE__, __LINE__);
85 
86  filename.insert(0, catalog->get_root() + "/"); // add root as a prefix to filename
87  BESDEBUG("besxml", "filename: " << filename << endl);
88 
89  ofstream ofs(filename.c_str(), ios::trunc|ios::binary);
90  if (!ofs.is_open())
91  throw BESSyntaxUserError(string("Build site map could not write to the site map file: ") + filename, __FILE__, __LINE__);
92 
93  catalog->get_site_map(prefix, node_suffix, leaf_suffix, ofs, "/");
94 
95  d_cmd_log_info = string("build site map for catalog '").append(catalog_name).append("' and write to '").append(filename);
96 
97  // Set action_name here because the NULLResponseHandler (aka NULL_ACTION) won't know
98  // which command used it (current ResponseHandlers set this because there is a 1-to-1
99  // correlation between XMLCommands and ResponseHanlders). jhrg 2/8/18
100  d_xmlcmd_dhi.action_name = SITE_MAP;
101  d_xmlcmd_dhi.action = NULL_ACTION;
102 
103  // Set the response handler for the action in the command's DHI using the value of
104  // the DHI.action field. And, as a bonus, copy the d_cmd_log_info into DHI.data[LOG_INFO]
105  // and if VERBOSE logging is on, log that the command has been parsed.
106  BESXMLCommand::set_response();
107 }
