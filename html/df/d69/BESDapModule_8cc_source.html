76 void BESDapModule::initialize(const string &modname)
77 {
78  BESDEBUG("dap", "Initializing DAP Modules:" << endl);
79 
80  BESDEBUG("dap", " adding " << modname << " request handler" << endl);
81  BESRequestHandlerList::TheList()->add_handler(modname, new BESDapRequestHandler(modname));
82 
83  BESDEBUG("dap", " adding " << DAS_RESPONSE << " response handler" << endl);
84  BESResponseHandlerList::TheList()->add_handler(DAS_RESPONSE, BESDASResponseHandler::DASResponseBuilder);
85 
86  BESDEBUG( "dap", " adding " << DDS_RESPONSE << " response handler" << endl );

88 
89  BESDEBUG("dap", " adding " << DDX_RESPONSE << " response handler" << endl);
90  BESResponseHandlerList::TheList()->add_handler(DDX_RESPONSE, BESDDXResponseHandler::DDXResponseBuilder);
91 
92  BESDEBUG("dap", " adding " << DATA_RESPONSE << " response handler" << endl);
93  BESResponseHandlerList::TheList()->add_handler(DATA_RESPONSE, BESDataResponseHandler::DataResponseBuilder);
94 
95  BESDEBUG("dap", " adding " << DATADDX_RESPONSE << " response handler" << endl);
96  BESResponseHandlerList::TheList()->add_handler(DATADDX_RESPONSE, BESDataDDXResponseHandler::DataDDXResponseBuilder);
97 
98  BESDEBUG("dap", " adding " << DMR_RESPONSE << " response handler" << endl);
99  BESResponseHandlerList::TheList()->add_handler(DMR_RESPONSE, BESDMRResponseHandler::DMRResponseBuilder);
100 
101  BESDEBUG("dap", " adding " << DAP4DATA_RESPONSE << " response handler" << endl);
102  BESResponseHandlerList::TheList()->add_handler(DAP4DATA_RESPONSE, BESDap4ResponseHandler::Dap4ResponseBuilder);
103 
104  BESDEBUG("dap", " adding " << CATALOG_RESPONSE << " response handler" << endl);
105  BESResponseHandlerList::TheList()->add_handler(CATALOG_RESPONSE, BESCatalogResponseHandler::CatalogResponseBuilder);
106 
107  BESDEBUG("dap", "Adding " << OPENDAP_SERVICE << " services:" << endl);
108  BESServiceRegistry *registry = BESServiceRegistry::TheRegistry();
109  registry->add_service(OPENDAP_SERVICE);
110  registry->add_to_service(OPENDAP_SERVICE, DAS_SERVICE, DAS_DESCRIPT, DAP2_FORMAT);
111  registry->add_to_service(OPENDAP_SERVICE, DDS_SERVICE, DDS_DESCRIPT, DAP2_FORMAT);
112  registry->add_to_service(OPENDAP_SERVICE, DDX_SERVICE, DDX_DESCRIPT, DAP2_FORMAT);
113  registry->add_to_service(OPENDAP_SERVICE, DATA_SERVICE, DATA_DESCRIPT, DAP2_FORMAT);
114  registry->add_to_service(OPENDAP_SERVICE, DATADDX_SERVICE, DATADDX_DESCRIPT, DAP2_FORMAT);
115 
116  registry->add_to_service(OPENDAP_SERVICE, DMR_SERVICE, DMR_DESCRIPT, DAP2_FORMAT);
117  registry->add_to_service(OPENDAP_SERVICE, DAP4DATA_SERVICE, DAP4DATA_DESCRIPT, DAP2_FORMAT);
118 
119  BESDEBUG("dap", "Initializing DAP Basic Transmitters:" << endl);
120  BESReturnManager::TheManager()->add_transmitter(DAP2_FORMAT, new BESDapTransmit());
121  // TODO ?? BESReturnManager::TheManager()->add_transmitter( DAP4_FORMAT, new BESDapTransmit( ) );
122 
123  BESDEBUG("dap", " adding dap exception handler" << endl);
124  BESExceptionManager::TheEHM()->add_ehm_callback(BESDapError::handleException);
125 
126 #if 0
127  // Aggregations are no longer run. jhrg 11/9/17

129  // TODO Add these names to BESDapNames.h
130  BESDEBUG("dap", " adding null aggregation handler" << endl);
131  BESAggFactory::TheFactory()->add_handler("null.aggregation", BESDapNullAggregationServer::NewBESDapNullAggregationServer);
132 #endif
133 #if 0
134  // Removed jhrg 3/17/15
135  BESAggFactory::TheFactory()->add_handler("sequence.aggregation", BESDapSequenceAggregationServer::NewBESDapSequenceAggregationServer);
136 #endif
137

139  WrapItUp *wiu = new WrapItUp();

141 
142  BESDEBUG("dap", " adding dap debug context" << endl);
143  BESDebug::Register("dap");
144 
145  BESDEBUG("dap", "Done Initializing DAP Modules:" << endl);

148 void BESDapModule::terminate(const string &modname)
149 {
150  BESDEBUG("dap", "Removing DAP Modules:" << endl);
151 
152  BESResponseHandlerList::TheList()->remove_handler(DAS_RESPONSE);
153  BESResponseHandlerList::TheList()->remove_handler(DDS_RESPONSE);
154  BESResponseHandlerList::TheList()->remove_handler(DDX_RESPONSE);
155  BESResponseHandlerList::TheList()->remove_handler(DATA_RESPONSE);
156  BESResponseHandlerList::TheList()->remove_handler(DATADDX_RESPONSE);
157  BESResponseHandlerList::TheList()->remove_handler(CATALOG_RESPONSE);
158 
159  BESResponseHandlerList::TheList()->remove_handler(DMR_RESPONSE);
160  BESResponseHandlerList::TheList()->remove_handler(DAP4DATA_RESPONSE);
161 
162  BESResponseHandlerList::TheList()->remove_handler(CATALOG_RESPONSE);
163 
164  BESDEBUG("dap", " removing " << OPENDAP_SERVICE << " services" << endl);
165  BESServiceRegistry::TheRegistry()->remove_service(OPENDAP_SERVICE);
166 
167  BESDEBUG("dap", " removing dap Request Handler " << modname << endl);
168  BESRequestHandler *rh = BESRequestHandlerList::TheList()->remove_handler(modname);
169  if (rh) delete rh;
170 
171  BESReturnManager::TheManager()->del_transmitter(DAP2_FORMAT);
172  // TODO ?? BESReturnManager::TheManager()->del_transmitter( DAP4_FORMAT );
173 
174 #if 0
175  BESAggFactory::TheFactory()->remove_handler("null.aggregation");
176 #endif
177 
178  BESDEBUG("dap", "Done Removing DAP Modules:" << endl);
179 }
