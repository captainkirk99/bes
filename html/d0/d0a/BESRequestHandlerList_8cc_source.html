306 void BESRequestHandlerList::dump(ostream &strm) const
307 {
308  strm << BESIndent::LMarg << "BESRequestHandlerList::dump - (" << (void *) this << ")" << endl;
309  BESIndent::Indent();
310  if (_handler_list.size()) {
311  strm << BESIndent::LMarg << "registered handlers:" << endl;
312  BESIndent::Indent();
313  BESRequestHandlerList::Handler_citer i = _handler_list.begin();
314  BESRequestHandlerList::Handler_citer ie = _handler_list.end();
315  for (; i != ie; i++) {
316  BESRequestHandler *rh = (*i).second;
317  rh->dump(strm);
318  }
319  BESIndent::UnIndent();
320  }
321  else {
322  strm << BESIndent::LMarg << "registered handlers: none" << endl;
