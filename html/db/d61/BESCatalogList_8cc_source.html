232 bool BESCatalogList::deref_catalog(const string &catalog_name) {

288 BESCatalogEntry *
289 BESCatalogList::show_catalogs(BESCatalogEntry *entry, bool show_default) {
290  BESCatalogEntry *myentry = entry;
291  if (!myentry) {
292  myentry = new BESCatalogEntry("/", "");
293  }
294  catalog_citer i = d_catalogs.begin();
295  catalog_citer e = d_catalogs.end();
296  for (; i != e; i++) {
297  // if show_default is true then display all catalogs
298  // if !show_default but this current catalog is not the default
299  // then display
300  if (show_default || (*i).first != default_catalog_name()) {
301  BESCatalog *catalog = (*i).second;
302  catalog->show_catalog("", myentry);
303  }
304  }
305 
306  return myentry;
307 }
