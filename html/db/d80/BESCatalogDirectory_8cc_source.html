101 BESCatalogEntry *
102 BESCatalogDirectory::show_catalog(const string &node, BESCatalogEntry *entry)
103 {
104  string use_node = node;
105  // use_node should only end in '/' if that's the only character in which
106  // case there's no need to call find()
107  if (!node.empty() && node != "/") {
108  string::size_type pos = use_node.find_last_not_of("/");
109  use_node = use_node.substr(0, pos + 1);
110  }
111 
112  // This takes care of bizarre cases like "///" where use_node would be
113  // empty after the substring call.
114  if (use_node.empty()) use_node = "/";
115 
116  string rootdir = d_utils->get_root_dir();
117  string fullnode = rootdir;
118  if (!use_node.empty()) {
119  // TODO It's hard to know just what this code is supposed to do, but I
120  // think the following can be an error. Above, if use_node is empty(), the use_node becomes
121  // "/" and then it's not empty() and fullnode becomes "<stuff>//" but we just
122  // jumped through all kinds of hoops to make sure there was either zero
123  // or one trailing slash. jhrg 2.26.18
124  fullnode = fullnode + "/" + use_node;
125  }
126 
127  string basename;
128  string::size_type slash = fullnode.rfind("/");
129  if (slash != string::npos) {
130  basename = fullnode.substr(slash + 1, fullnode.length() - slash);
131  }
132  else {
133  basename = fullnode;
134  }
135 
136  // fullnode is the full pathname of the node, including the 'root' pathanme
137  // basename is the last component of fullnode
138 
139  BESDEBUG("bes",
140  "BESCatalogDirectory::show_catalog: " << "use_node = " << use_node << endl << "rootdir = " << rootdir << endl << "fullnode = " << fullnode << endl << "basename = " << basename << endl);
141 
142  // This will throw the appropriate exception (Forbidden or Not Found).
143  // Checks to make sure the different elements of the path are not
144  // symbolic links if follow_sym_links is set to false, and checks to
145  // make sure have permission to access node and the node exists.
146  // TODO Move up; this can be done once use_node is set. jhrg 2.26.18
147  BESUtil::check_path(use_node, rootdir, d_utils->follow_sym_links());
148 
149  // If null is passed in, then return the new entry, else add the new entry to the
150  // existing Entry object. jhrg 2.26.18
151  BESCatalogEntry *myentry = new BESCatalogEntry(use_node, get_catalog_name());
152  if (entry) {
153  // if an entry was passed, then add this one to it
154  entry->add_entry(myentry);
155  }
156  else {
157  // else we want to return the new entry created
158  entry = myentry;
159  }
160 
161  // Is this node a directory?
162  // TODO use stat() instead. jhrg 2.26.18
163  DIR *dip = opendir(fullnode.c_str());
164  if (dip != NULL) {
165  try {
166  // The node is a directory
167 
168  // if the directory requested is in the exclude list then we won't
169  // let the user see it.
170  if (d_utils->exclude(basename)) {
171  string error = "You do not have permission to view the node " + use_node;
172  throw BESForbiddenError(error, __FILE__, __LINE__);
173  }
174 
175  // Now that we are ready to start building the response data we
176  // cancel any pending timeout alarm according to the configuration.
177  BESUtil::conditional_timeout_cancel();
178 
179  bool dirs_only = false;
180  // TODO This is the only place in the code where get_entries() is called
181  // jhrg 2.26.18
182  d_utils->get_entries(dip, fullnode, use_node, myentry, dirs_only);
183  }
184  catch (... /*BESError &e */) {
185  closedir(dip);
186  throw /* e */;
187  }
188  closedir(dip);
189 
190  // TODO This is the only place this method is called. replace the static method
191  // with an object call (i.e., d_utils)? jhrg 2.26.18
192  BESCatalogUtils::bes_add_stat_info(myentry, fullnode);
193  }
194  else {
195  // if the node is not in the include list then the requester does
196  // not have access to that node
197  if (d_utils->include(basename)) {
198  struct stat buf;
199  int statret = 0;
200  if (d_utils->follow_sym_links() == false) {
201  /*statret =*/(void) lstat(fullnode.c_str(), &buf);
202  if (S_ISLNK(buf.st_mode)) {
203  string error = "You do not have permission to access node " + use_node;
204  throw BESForbiddenError(error, __FILE__, __LINE__);
205  }
206  }
207  statret = stat(fullnode.c_str(), &buf);
208  if (statret == 0 && S_ISREG(buf.st_mode)) {
209  BESCatalogUtils::bes_add_stat_info(myentry, fullnode);
210 
211  list<string> services;
212  BESCatalogUtils::isData(node, get_catalog_name(), services);
213  myentry->set_service_list(services);
214  }
215  else if (statret == 0) {
216  string error = "You do not have permission to access " + use_node;
217  throw BESForbiddenError(error, __FILE__, __LINE__);
218  }
219  else {
220  // ENOENT means that the path or part of the path does not
221  // exist
222  if (errno == ENOENT) {
223  string error = "Node " + use_node + " does not exist";
224  char *s_err = strerror(errno);
225  if (s_err) {
226  error = s_err;
227  }
228  throw BESNotFoundError(error, __FILE__, __LINE__);
229  }
230  // any other error means that access is denied for some reason
231  else {
232  string error = "Access denied for node " + use_node;
233  char *s_err = strerror(errno);
234  if (s_err) {
235  error = error + s_err;
236  }
237  throw BESNotFoundError(error, __FILE__, __LINE__);
238  }
239  }
240  }
241  else {
242  string error = "You do not have permission to access " + use_node;
243  throw BESForbiddenError(error, __FILE__, __LINE__);
244  }
245  }
246 
247  return entry;
248 }

270 {
271  char buf[sizeof "YYYY-MM-DDTHH:MM:SSzone"];
272  int status = 0;
273 
274  // From StackOverflow:
275  // This will work too, if your compiler doesn't support %F or %T:
276  // strftime(buf, sizeof buf, "%Y-%m-%dT%H:%M:%S%Z", gmtime(&now));

279  // local time using BES.LogTimeLocal=yes in bes.conf. jhrg 11/15/17
280  if (!use_local_time)
281  status = strftime(buf, sizeof buf, "%FT%T%Z", gmtime(&the_time));

283  status = strftime(buf, sizeof buf, "%FT%T%Z", localtime(&the_time));
284 
285  if (!status)
286  LOG("Error getting last modified time time for a leaf item in BESCatalogDirectory.");
287 
288  return buf;
289 }

314 CatalogNode *
315 BESCatalogDirectory::get_node(const string &path) const
316 {
317  if (path[0] != '/') throw BESInternalError("Catalog paths must start with a slash (/)", __FILE__, __LINE__);
318 
319  string rootdir = d_utils->get_root_dir();
320 
321  // This will throw the appropriate exception (Forbidden or Not Found).
322  // Checks to make sure the different elements of the path are not
323  // symbolic links if follow_sym_links is set to false, and checks to
324  // make sure have permission to access node and the node exists.
325  BESUtil::check_path(path, rootdir, d_utils->follow_sym_links());
326 
327  string fullpath = rootdir + path;
328 
329  DIR *dip = opendir(fullpath.c_str());
330  if (!dip)
331  throw BESInternalError(
332  "A BESCatalogDirectory can only return nodes for directory. The path '" + path
333  + "' is not a directory for BESCatalog '" + get_catalog_name() + "'.", __FILE__, __LINE__);
334 
335  try {
336  // The node is a directory
337 
338  // Based on other code (show_catalogs()), use BESCatalogUtils::exclude() on
339  // a directory, but BESCatalogUtils::include() on a file.
340  if (d_utils->exclude(path))
341  throw BESForbiddenError(
342  string("The path '") + path + "' is not included in the catalog '" + get_catalog_name() + "'.",
343  __FILE__, __LINE__);
344 
345  CatalogNode *node = new CatalogNode(path);
346 
347  node->set_catalog_name(get_catalog_name());
348  struct stat buf;
349  int statret = stat(fullpath.c_str(), &buf);
350  if (statret == 0 /* && S_ISDIR(buf.st_mode) */)
351  node->set_lmt(get_time(buf.st_mtime));
352 
353  struct dirent *dit;
354  while ((dit = readdir(dip)) != NULL) {
355  string item = dit->d_name;
356  if (item == "." || item == "..") continue;
357 
358  string item_path = fullpath + "/" + item;
359 
360  // TODO add a test in configure for the readdir macro(s) DT_REG, DT_LNK
361  // and DT_DIR and use those, if present, to determine if the name is a
362  // link, directory or regular file. These are not present on all systems.
363  // Also, since we need mtime, this is not a huge time saver. But if we
364  // decide not to use the mtime, using these macros could save lots of system
365  // calls. jhrg 3/9/18
366 
367  // Skip this dir entry if it is a sym link and follow links is false
368  if (d_utils->follow_sym_links() == false) {
369  struct stat lbuf;
370  (void) lstat(item_path.c_str(), &lbuf);
371  if (S_ISLNK(lbuf.st_mode)) continue;
372  }
373 
374  // Is this a directory or a file? Should it be excluded or included?
375  statret = stat(item_path.c_str(), &buf);
376  if (statret == 0 && S_ISDIR(buf.st_mode) && !d_utils->exclude(item)) {
377 #if 0
378  // Add a new node; set the size to zero.
379  node->add_item(new CatalogItem(item, 0, get_time(buf.st_mtime), CatalogItem::node));
380 #endif
381  node->add_node(new CatalogItem(item, 0, get_time(buf.st_mtime), CatalogItem::node));
382  }
383  else if (statret == 0 && S_ISREG(buf.st_mode) && d_utils->include(item)) {
384 #if 0
385  // Add a new leaf.
386  node->add_item(new CatalogItem(item, buf.st_size, get_time(buf.st_mtime),
387  d_utils->is_data(item), CatalogItem::leaf));
388 #endif
389  node->add_leaf(new CatalogItem(item, buf.st_size, get_time(buf.st_mtime),
390  d_utils->is_data(item), CatalogItem::leaf));
391  }
392  else {
393  VERBOSE("Excluded the item '" << item_path << "' from the catalog '" << get_catalog_name() << "' node listing.");
394  }
395  } // end of the while loop
396 
397  closedir(dip);
398 
399  CatalogItem::CatalogItemAscending ordering;
400 
401  sort(node->nodes_begin(), node->nodes_end(), ordering);
402  sort(node->leaves_begin(), node->leaves_end(), ordering);
403 
404  return node;
405  }
406  catch (...) {
407  closedir(dip);
408  throw;
409  }
410 }

431 void BESCatalogDirectory::get_site_map(const string &prefix, const string &node_suffix, const string &leaf_suffix,
432  ostream &out, const string &path) const
433 {
434  auto_ptr<CatalogNode> node(get_node(path));
435 
436 #if ITEMS
437  for (CatalogNode::item_citer i = node->items_begin(), e = node->items_end(); i != e; ++i) {
438  if ((*i)->get_type() == CatalogItem::leaf && (*i)->is_data()) {
439  out << prefix << path << (*i)->get_name() << leaf_suffix << endl;
440  }
441  else if ((*i)->get_type() == CatalogItem::node) {
442  get_site_map(prefix, leaf_suffix, out, path + (*i)->get_name() + "/");
443  }
444  }
445 #endif
446 
447  if (!node_suffix.empty())
448  out << prefix << path << node_suffix << endl;
449 
450  // Depth-first node traversal. Assume the nodes and leaves are sorted
451  for (CatalogNode::item_citer i = node->nodes_begin(), e = node->nodes_end(); i != e; ++i) {
452  assert((*i)->get_type() == CatalogItem::node);
453  get_site_map(prefix, node_suffix, leaf_suffix, out, path + (*i)->get_name() + "/");
454  }
455 
456  // For leaves, only write the data items
457  for (CatalogNode::item_citer i = node->leaves_begin(), e = node->leaves_end(); i != e; ++i) {
458  assert((*i)->get_type() == CatalogItem::leaf);
459  if ((*i)->is_data() && !leaf_suffix.empty())
460  out << prefix << path << (*i)->get_name() << leaf_suffix << endl;
461  }
462 }
