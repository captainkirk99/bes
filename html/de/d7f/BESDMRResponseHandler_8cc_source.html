74 void BESDMRResponseHandler::execute(BESDataHandlerInterface &dhi)
75 {
76  dhi.action_name = DMR_RESPONSE_STR;
77 
78  bool xml_base_found = false;
79  string xml_base = BESContextManager::TheManager()->get_context("xml:base", xml_base_found);
80 
81  // Look in the MDS for dhi.container.get_real_name().
82  // if found, use that response, else build it.
83  // If the MDS is disabled, don't use it.
84  GlobalMetadataStore *mds = GlobalMetadataStore::get_instance();
85 
86  GlobalMetadataStore::MDSReadLock lock;
87 
88  dhi.first_container();
89  if (mds) lock = mds->is_dmr_available(dhi.container->get_real_name());
90 
91  if (mds && lock() && dhi.container->get_dap4_constraint().empty()) { // no CE
92  // FIXME Does not work for constrained DMR requests
93  BESDEBUG("dmr", __func__ << " Locked: " << dhi.container->get_real_name() << endl);
94  // send the response
95  mds->get_dmr_response(dhi.container->get_real_name(), dhi.get_output_stream());
96  // suppress transmitting a ResponseObject in transmit()
97  d_response_object = 0;
98  }
99 #if DMR_CE
100  // this shows how to support DMR/DDS requests with CEs. I have no tests for this,
101  // however. And there are other things that are breaking still... jhrg 3/19/18
102  else if (mds && lock() && !dhi.container->get_dap4_constraint().empty()) { // with CE
103  stringstream oss;
104  mds->get_dmr_response(dhi.container->get_real_name(), oss);
105 
106  DMR *dmr = new DMR(new D4BaseTypeFactory, "mds");
107 
108  if (xml_base_found && !xml_base.empty()) dmr->set_request_xml_base(xml_base);
109 
110  D4ParserSax2 parser;
111  parser.intern(oss.str(), dmr);
112 
113  // Do not leak the BaseType factory
114  delete dmr->factory();
115  dmr->set_factory(0);
116 
117  // The transmit() method will evaluate the CE before sending the response.
118  d_response_object = new BESDMRResponse(dmr);
119  }
120 #endif
121  else { // no MDS
122  DMR *dmr = new DMR();
123 
124  if (xml_base_found && !xml_base.empty()) dmr->set_request_xml_base(xml_base);
125 
126  d_response_object = new BESDMRResponse(dmr);
127 
128  BESRequestHandlerList::TheList()->execute_each(dhi);
129 
130  if (mds) {
131  dhi.first_container(); // must reset container; execute_each() iterates over all of them
132  BESDEBUG("dmr", __func__ << " Storing: " << dhi.container->get_real_name() << endl);
133  mds->add_responses(static_cast<BESDMRResponse*>(d_response_object)->get_dmr(),
134  dhi.container->get_real_name());
135  }
136  }
137 }
